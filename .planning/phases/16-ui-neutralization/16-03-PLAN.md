          <span>{{ translationService.translate('nav.knowledge-base') }}</span>
        </a>

          <span>{{ translationService.translate('nav.ai-assistant') }}</span>

          <lucide-icon name="file-text"></lucide-icon>
          <span>{{ translationService.translate('nav.documents') }}</span>
        </a>
      </div>
    </section>

    <!-- Role-Adaptive Sections -->
    @if (authService.isAdmin() || authService.isAgent()) {
      <section class="role-actions">
        @if (authService.isAdmin()) {
          <a routerLink="/workspace/settings" class="role-link">
            {{ translationService.translate('nav.workspace-settings') }}

          <a routerLink="/tickets/queue" class="role-link">
            {{ translationService.translate('nav.ticket-queue') }}
          </a>
        }
      </section>
    }

    <!-- Recent Activity (placeholder) -->
    <section class="recent-activity">
      <h3>{{ translationService.translate('dashboard.recent-activity') }}</h3>
      <p class="text-muted">{{ translationService.translate('dashboard.recent-empty') }}</p>
    </section>
